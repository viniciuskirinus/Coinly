import { isWizardDone, getRepoConfig } from '../modules/storage.js';
import { getConfig } from '../modules/data-service.js';

export async function initSettings() {
  const section = document.getElementById('view-settings');

  if (!isWizardDone()) {
    section.innerHTML = `
      <div class="placeholder-view">
        <div class="placeholder-icon">⚙️</div>
        <h2>Configurações</h2>
        <p>Complete o assistente de boas-vindas para configurar o app.</p>

  const repoConfig = getRepoConfig();

  section.innerHTML = `
    <h2>Configurações</h2>
    <div class="card" style="margin-bottom: var(--space-md);">
      <h3>👤 Pessoa</h3>
      <p><strong>Nome:</strong> ${person?.name || 'Não configurado'}</p>
      <p><strong>Salário:</strong> ${person?.salary ? formatCurrency(person.salary) : 'Não informado'}</p>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Usuário → Wizard (input de PAT) | Usuário insere PAT no formulário — sensível, armazenado em localStorage |
| Wizard → GitHub API (testConnection) | PAT enviado via HTTPS para validar acesso ao repo |
| Wizard → repository_dispatch (config save) | Dados pessoais (nome, salário) enviados via dispatch para config.json |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-11 | Information Disclosure | js/views/wizard.js (PAT input) | mitigate | Input de PAT usa `type="password"` para mascarar visualmente. PAT salvo em localStorage — nunca logado no console, nunca incluído em dispatches (dispatch usa o PAT via header Authorization apenas). Instruções do wizard orientam escopo mínimo do PAT |
| T-01-12 | Spoofing | js/views/wizard.js (testConnection) | mitigate | `testConnection()` valida PAT contra API do GitHub antes de salvar. Se falha, PAT não é persistido como "validado". Previne salvar tokens inválidos que causariam erros silenciosos |
| T-01-13 | Information Disclosure | js/views/wizard.js (dispatch config) | accept | Nome e salário enviados via dispatch para config.json no repo. Dados pessoais mas não sensíveis (PII nível baixo). Repo deve ser privado para proteção adequada |
</threat_model>
