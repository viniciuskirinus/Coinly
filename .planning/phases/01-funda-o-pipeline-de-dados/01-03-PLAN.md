**1. `js/modules/github-api.js`** — Client repository_dispatch. Seguir EXATAMENTE o padrão "Pattern 2: GitHub API Client" do RESEARCH.md:

```javascript
import { getRepoConfig } from './storage.js';

export async function dispatch(eventType, data, target) {
  const { owner, repo, pat } = getRepoConfig();
  if (!pat) throw new Error('PAT não configurado');

  const resp = await fetch(
    `https://api.github.com/repos/${owner}/${repo}/dispatches`,
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${pat}`,
        'Accept': 'application/vnd.github+json',
        'X-GitHub-Api-Version': '2022-11-28'
      },
      body: JSON.stringify({
        event_type: eventType,
        client_payload: {
          action: eventType,
          target: target || null,
          data: data,
          timestamp: new Date().toISOString()
        }
      })
    }
  );

  if (resp.status === 204) return { success: true };
  if (resp.status === 404) return { success: false, error: 'Repo não encontrado ou PAT sem permissão' };
  if (resp.status === 422) return { success: false, error: 'Payload inválido' };
  return { success: false, error: `HTTP ${resp.status}` };
}

export async function testConnection() {
  const { owner, repo, pat } = getRepoConfig();
  if (!owner || !repo || !pat) return { success: false, error: 'Configuração incompleta' };

  try {
    const resp = await fetch(
      `https://api.github.com/repos/${owner}/${repo}`,
      {
        headers: {
          'Authorization': `Bearer ${pat}`,
          'Accept': 'application/vnd.github+json',
          'X-GitHub-Api-Version': '2022-11-28'
        }
      }
    );
    if (resp.ok) return { success: true };
    if (resp.status === 401) return { success: false, error: 'Token inválido' };
    if (resp.status === 404) return { success: false, error: 'Repo não encontrado ou sem permissão' };
    return { success: false, error: `HTTP ${resp.status}` };
  } catch (e) {
    return { success: false, error: 'Falha na conexão: ' + e.message };
  }
}
```

Pontos críticos:
- O `client_payload` usa pattern de nesting: 4 campos top-level (`action`, `target`, `data`, `timestamp`) — dentro do limite de 10 (per RESEARCH.md Pattern 4)
- `target` é usado para indicar o arquivo alvo (ex: `"2026-04"` para transações do mês)
- `testConnection()` é para o wizard verificar se o PAT funciona (GET no repo, não dispatch)
- API version header: `2022-11-28` (per GitHub Docs)

**1. `.github/scripts/process-write.mjs`** — Script Node.js que processa escritas. Usar extensão `.mjs` para ES Modules sem necessidade de package.json (per RESEARCH.md Assumptions Log A1).

Variáveis de ambiente disponíveis (passadas pelo workflow):
- `process.env.EVENT_TYPE` — string com o tipo do evento (ex: `create-transaction`)
- `process.env.PAYLOAD` — JSON string do `client_payload` completo

Estrutura do script:

```javascript
import { readFileSync, writeFileSync, existsSync, mkdirSync } from 'fs';

const eventType = process.env.EVENT_TYPE;
const payload = JSON.parse(process.env.PAYLOAD);
const data = payload.data;
const target = payload.target;

console.log(`Processing: ${eventType}`);
console.log(`Timestamp: ${payload.timestamp}`);

const handlers = {
  'create-transaction': handleCreateTransaction,
  'edit-transaction': handleEditTransaction,
  'delete-transaction': handleDeleteTransaction,
  'update-config': handleUpdateConfig,
  'update-categories': handleUpdateCategories,
  'update-payment-methods': handleUpdatePaymentMethods
};

const handler = handlers[eventType];
if (!handler) {
  console.error(`Unknown event type: ${eventType}`);
  process.exit(1);
}

handler(data, target);
console.log('Done.');
```

**Handlers implementados:**

**`handleCreateTransaction(txn, yearMonth)`:**
- `yearMonth` vem do campo `target` do payload (ex: `"2026-04"`)
- Se `yearMonth` não fornecido, extrair de `txn.date.slice(0, 7)`
- Caminho: `data/transactions/${yearMonth}.json`
- Se arquivo não existe: criar com `{ _schema_version: 1, month: yearMonth, lastId: 0, transactions: [] }`
- Se pasta não existe: `mkdirSync('data/transactions', { recursive: true })`
- Incrementar `lastId`, atribuir `txn.id = file.lastId`
- Atribuir `txn.createdAt = new Date().toISOString()`
- Push `txn` no array `transactions`
- Escrever com `JSON.stringify(file, null, 2) + '\n'`
- Log: `Created transaction ${txn.id} in ${yearMonth}`

**`handleEditTransaction(txn, yearMonth)`:**
- Ler arquivo de `yearMonth`
- Encontrar transação por `txn.id`
- Se não encontrada: `console.error()` + `process.exit(1)`
- Atualizar campos (merge: `Object.assign(existing, txn)`)
- Atribuir `existing.updatedAt = new Date().toISOString()`
- Escrever arquivo atualizado

**`handleDeleteTransaction(payload, yearMonth)`:**
- `payload` pode ter `id` (single delete) ou `ids` (bulk delete)
- Filtrar array `transactions` removendo os IDs
- Log quantidade removida
- Escrever arquivo atualizado

**`handleUpdateConfig(newConfig)`:**
- Ler `data/config.json`
- Merge: `Object.assign(file, newConfig)`
- Atribuir `file.updatedAt = new Date().toISOString()`
- Escrever arquivo atualizado

**`handleUpdateCategories(categories)`:**
- Escrever `data/categories.json` com o objeto `categories` completo
- Manter `_schema_version: 1`

**`handleUpdatePaymentMethods(methods)`:**
- Escrever `data/payment-methods.json` com `{ _schema_version: 1, methods: methods.methods || methods }`

**Validação de input:**
- Antes de cada handler: verificar que `data` não é null/undefined
- Em `handleCreateTransaction`: verificar que `txn.date` existe e é string no formato YYYY-MM-DD
- Em `handleEditTransaction`: verificar que `txn.id` existe e é number
- Se validação falha: `console.error('Validation failed: ...')` + `process.exit(1)`

**Helper para ler JSON file:**
```javascript
function readJSONFile(path) {
  if (!existsSync(path)) return null;
  return JSON.parse(readFileSync(path, 'utf-8'));
}

function writeJSONFile(path, data) {
  writeFileSync(path, JSON.stringify(data, null, 2) + '\n', 'utf-8');
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → GitHub API | Frontend envia PAT via HTTPS ao GitHub. PAT em localStorage é o elo mais fraco |
| GitHub Action runner → repo data/ | Script processa payload não-confiável (origin: client_payload do dispatch) e escreve em arquivos do repo |
| Push to main → GitHub Pages CDN | Deploy automático — qualquer push ao main atualiza o site público |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-06 | Tampering | .github/scripts/process-write.mjs | mitigate | Validar schema do payload antes de escrever: verificar campos obrigatórios (date, amount, type para transações), rejeitar payloads com campos inesperados. `JSON.parse()` com try/catch protege contra JSON malformado |
| T-01-07 | Tampering | .github/workflows/write-data.yml | mitigate | NUNCA usar `${{ github.event.client_payload.* }}` em blocos `run:`. Payload passado EXCLUSIVAMENTE via `env: PAYLOAD: ${{ toJSON(github.event.client_payload) }}`. Event type via `env: EVENT_TYPE: ${{ github.event.action }}` |
| T-01-08 | Information Disclosure | js/modules/github-api.js | mitigate | PAT enviado apenas via header `Authorization: Bearer`. Fine-Grained PAT com escopo mínimo: `Contents: Read and write` scoped ao repo específico. Nunca logar o PAT no console |
| T-01-09 | Denial of Service | .github/workflows/write-data.yml | accept | Concurrency group limita a 1 running + 1 pending. Terceiro dispatch concorrente substitui o pending. Para uso pessoal (1-2 users), colisões improváveis. Frontend deve implementar debounce (3-5s) em fases futuras |
| T-01-10 | Elevation of Privilege | .github/workflows/write-data.yml | mitigate | Workflow tem `permissions: contents: write` apenas. Não tem `admin`, `security_events`, `actions: write`. PAT do usuário é Fine-Grained scoped ao repo |
</threat_model>
